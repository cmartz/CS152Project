Comp: EQ
      | NEQ
      | LT
      | GT
      | LTE
      | GTE
      ;

Expr: Mult_expr Expr_seq
      ;

Expr_seq: ADD Mult_expr Expr_seq
          | SUB Mult_expr Expr_seq
          |
          ;

Mult_expr: Term Mult_expr_seq
            ;

Mult_expr_seq: MULT Term Mult_expr_seq
               | DIV Term Mult_expr_seq
               | MOD Term Mult_expr_seq
               |
               ;
